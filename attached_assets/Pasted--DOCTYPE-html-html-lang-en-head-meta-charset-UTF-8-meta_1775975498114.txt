    const latH=await rpc('eth_blockNumber');

    const logs=await rpc('eth_getLogs',[{address:ARENA_BATTLE,fromBlock:from,toBlock:'latest'}]);
    const stats={};
    for(const log of logs){
      if(log.topics.length<3)continue;
      const w='0x'+log.topics[1].slice(26),l='0x'+log.topics[2].slice(26);
      if(!stats[w])stats[w]={wins:0,losses:0};
      if(!stats[l])stats[l]={wins:0,losses:0};
      stats[w].wins++;stats[l].losses++;
    }
    const entries=Object.entries(stats);
    const ranked=entries.length>0
      ?entries.map(([a,s])=>({addr:a,wins:s.wins,losses:s.losses,rate:s.wins/(s.wins+s.losses)})).sort((a,b)=>b.wins-a.wins).slice(0,20)

    renderPodium(ranked);renderLbTable(ranked);
    document.getElementById('lb-status').textContent=(entries.length>0?'Live data':'Demo data')+' · '+new Date().toLocaleTimeString([],{hour:'2-digit',minute:'2-digit'});
  }catch{const d=demoBoard();renderPodium(d);renderLbTable(d);document.getElementById('lb-status').textContent='Chain unavailable · demo';}
  finally{btn.classList.remove('spinning');}
}

async function resolveOwned(ids){
  nftOwned=new Set();
  for(let i=0;i<ids.length;i+=8){
    const chunk=ids.slice(i,i+8);
    const res=await Promise.allSettled(chunk.map(id=>call(ARENA_CHAMPION,SIG.ownerOf+eU256(id))));
    res.forEach((r,j)=>{if(r.status==='fulfilled'){const own='0x'+r.value.slice(-40);if(own.toLowerCase()===walletAddr.toLowerCase())nftOwned.add(chunk[j]);}});
  }
}
async function fetchMeta(id){
  try{
    const h=await call(ARENA_CHAMPION,SIG.tokenURI+eU256(id));
    const uri=dStr(h);if(!uri||uri==='—')return null;
    if(uri.startsWith('data:application/json;base64,'))return JSON.parse(atob(uri.slice(29)));
    const url=uri.replace('ipfs://','https://ipfs.io/ipfs/');
    return await(await fetch(url)).json();
  }catch{return null;}
}

const SEV_KW={critical:['selfdestruct','delegatecall','proxy'],high:['reentrancy','overflow','tx.origin','access'],medium:['unchecked','timestamp','oracle'],low:['gas','emit','visibility','natspec']};
const DET={critical:'Selfdestruct / Proxy Risk',high:'Reentrancy / Access Control',medium:'Unchecked / Oracle',low:'Gas / Visibility',info:'Static Analysis Pass'};
function gSev(txt){const t=txt.toLowerCase();for(const[s,kw]of Object.entries(SEV_KW))if(kw.some(k=>t.includes(k)))return s;return'info';}
async function loadAuditFeed(){
  try{
    const[cR,iR]=await Promise.all([fetch(`https://api.github.com/repos/${AUDIT_REPO}/commits?per_page=20`),fetch(`https://api.github.com/repos/${AUDIT_REPO}/issues?state=all&per_page=10`)]);
    const commits=cR.ok?await cR.json():[],issues=iR.ok?await iR.json():[];
    const items=[];
    for(const c of commits.slice(0,12)){const msg=c.commit.message.split('\n')[0],sev=gSev(msg);items.push({sev,det:DET[sev],repo:AUDIT_REPO,desc:msg.slice(0,120),ts:new Date(c.commit.author.date).getTime(),url:c.html_url,ref:c.sha.slice(0,7)});}
    for(const iss of issues){const sev=gSev(iss.title);items.push({sev,det:DET[sev],repo:AUDIT_REPO,desc:iss.title.slice(0,120),ts:new Date(iss.created_at).getTime(),url:iss.html_url,ref:'#'+iss.number});}
    items.sort((a,b)=>b.ts-a.ts);
    renderFeed(items.slice(0,12));renderFeedStats(items);
  }catch{renderDemoFeed();}
}
function renderFeed(items){
  const f=document.getElementById('audit-feed');
  if(!items.length){f.innerHTML=`<div class="lb-empty">No events found</div>`;return;}
  f.innerHTML=items.map(it=>`<div class="audit-entry"><div class="audit-sev sev-${it.sev}">${it.sev.toUpperCase()}</div><div><div class="audit-detector">${it.det}</div><div class="audit-repo">${it.repo}</div><div class="audit-desc">${it.desc}</div></div><div class="audit-meta"><div class="audit-time">${ago(it.ts)}</div><a href="${it.url}" target="_blank" rel="noopener"><div class="audit-pr">${it.ref} ↗</div></a></div></div>`).join('');
}
function renderFeedStats(items){
  document.getElementById('as-total').textContent=items.length;
  document.getElementById('as-critical').textContent=items.filter(i=>i.sev==='critical').length;
  document.getElementById('as-high').textContent=items.filter(i=>i.sev==='high').length;
  document.getElementById('as-clean').textContent=items.filter(i=>i.sev==='info'||i.sev==='low').length;
}

async function connectWallet(){
  if(!window.ethereum){showToast('⚠️ MetaMask not detected');return;}
  try{
    const prov=new ethers.BrowserProvider(window.ethereum);
    await prov.send('eth_requestAccounts',[]);
    const signer=await prov.getSigner(),addr=await signer.getAddress(),net=await prov.getNetwork();
    walletAddr=addr;walletProv=prov;
    const btn=document.getElementById('wallet-btn');btn.textContent=sAddr(addr);btn.classList.add('connected');
    document.getElementById('wp-addr').textContent=sAddr(addr);
    const onBase=net.chainId===8453n;
    const ne=document.getElementById('wp-net');ne.textContent=onBase?'Base Mainnet ✓':`Wrong network (${net.chainId})`;ne.style.color=onBase?'var(--green)':'var(--red)';
    const eb=await prov.getBalance(addr);document.getElementById('wp-eth').textContent=parseFloat(ethers.formatEther(eb)).toFixed(4)+' ETH';
    try{const bh=await call(ARENA_TOKEN,SIG.balanceOf+eAddr(addr));arenaBal=dUint(bh);document.getElementById('wp-arena').textContent=fmtTok(arenaBal,18)+' ARENA';}catch{document.getElementById('wp-arena').textContent='—';}
    updateProUI();
    if(nftAll.length){await resolveOwned(nftAll);renderNftPage();}
    showToast('✅ Connected: '+sAddr(addr));
    if(!onBase)showToast('⚠️ Switch to Base Mainnet',4500);
    window.ethereum.on('accountsChanged',a=>{if(!a.length)disconnectWallet();else{walletAddr=null;connectWallet();}});
    window.ethereum.on('chainChanged',()=>location.reload());
  }catch(e){if(e.code!==4001)showToast('⚠️ Connection failed');}
}
